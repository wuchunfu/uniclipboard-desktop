1. **Create SetupAssemblyPorts struct in assembly.rs**:

**CRITICAL**: `SetupAssemblyPorts` bundles only the network/external adapter ports that the caller provides. It does NOT contain shared state (`emitter_cell`, `lifecycle_status`, `clipboard_integration_mode`). Those shared values are created by `with_setup()` and passed as separate parameters to `build_setup_orchestrator()`, ensuring ONE source of truth shared between the orchestrator and CoreRuntime.

**CRITICAL (watcher_control type)**: `watcher_control` MUST be typed as `Arc<dyn uc_platform::ports::WatcherControlPort>` (the raw platform port), NOT `Arc<dyn StartClipboardWatcherPort>` (the use-case wrapper). The wrapping from raw `WatcherControlPort` to `StartClipboardWatcherPort` requires `ClipboardIntegrationMode`, which is shared state created inside `with_setup()`. The caller (main.rs) already has the raw `WatcherControlPort` and should not need to resolve `clipboard_integration_mode` to construct `SetupAssemblyPorts`. The wrapping happens inside `build_setup_orchestrator()`.

```rust
use uc_app::usecases::{
```

```rust
/// Build the SetupOrchestrator with all required adapters.
///
/// This is the single composition point for SetupOrchestrator (RNTM-05).
/// Network/external adapters come via `ports`. Shared state (emitter_cell,
/// lifecycle_status, clipboard_integration_mode) and with_setup()-constructed
/// adapters (session_ready_emitter) come as separate parameters — the caller
/// (with_setup) passes the SAME instances to both this function and
/// CoreRuntime::new(), guaranteeing a single source of truth and preventing
/// the stale-emitter bug.
///
/// `clipboard_integration_mode` is needed to wrap the raw `WatcherControlPort`
/// (from ports) into `StartClipboardWatcherPort` (the use-case wrapper).
/// `session_ready_emitter` is constructed from `app_handle` in with_setup()
/// because TauriSessionReadyEmitter is a Tauri adapter that cannot be built
/// by the external caller (app_handle is created inside with_setup).
pub fn build_setup_orchestrator(
    deps: &AppDeps,
    ports: SetupAssemblyPorts,
    lifecycle_status: Arc<dyn LifecycleStatusPort>,
    emitter_cell: Arc<std::sync::RwLock<Arc<dyn HostEventEmitterPort>>>,
    clipboard_integration_mode: ClipboardIntegrationMode,
    session_ready_emitter: Arc<dyn SessionReadyEmitter>,
) -> Arc<SetupOrchestrator> {
    let initialize_encryption = Arc::new(uc_app::usecases::InitializeEncryption::from_ports(
        deps.security.encryption.clone(),
        deps.security.key_material.clone(),
        deps.security.key_scope.clone(),
        deps.security.encryption_state.clone(),
        deps.security.encryption_session.clone(),
    ));
    let mark_setup_complete = Arc::new(uc_app::usecases::MarkSetupComplete::from_ports(
        deps.setup_status.clone(),
    ));
    // Wrap raw WatcherControlPort into StartClipboardWatcherPort using clipboard_integration_mode.
    // This wrapping happens HERE (not in the caller) because clipboard_integration_mode is
    // shared state created by with_setup(), not a caller-provided external port.
    let start_watcher: Arc<dyn StartClipboardWatcherPort> =
        Arc::new(uc_platform::usecases::StartClipboardWatcher::from_port(
            ports.watcher_control,          // raw WatcherControlPort from caller
            clipboard_integration_mode,     // shared state from with_setup()
        ));
    let start_network = Arc::new(uc_app::usecases::StartNetworkAfterUnlock::from_port(
        deps.network_control.clone(),
    ));
    let app_lifecycle = Arc::new(uc_app::usecases::AppLifecycleCoordinator::from_deps(
        uc_app::usecases::AppLifecycleCoordinatorDeps {
            watcher: start_watcher,
            network: start_network,
            announcer: ports.device_announcer,
            emitter: session_ready_emitter, // <-- from separate parameter, NOT ports
            status: lifecycle_status,       // <-- from separate parameter, NOT ports
            lifecycle_emitter: ports.lifecycle_emitter,
        },
    ));
    let crypto_factory = Arc::new(uc_app::usecases::space_access::DefaultSpaceAccessCryptoFactory::new(
        deps.security.encryption.clone(),
        deps.security.key_material.clone(),
        deps.security.key_scope.clone(),
        deps.security.encryption_state.clone(),
        deps.security.encryption_session.clone(),
    ));
    let transport_port: Arc<Mutex<dyn SpaceAccessTransportPort>> =
        Arc::new(Mutex::new(uc_app::usecases::space_access::SpaceAccessNetworkAdapter::new(
            deps.network_ports.pairing.clone(),
            ports.space_access_orchestrator.context(),
        )));
    let proof_port: Arc<dyn uc_core::ports::space::ProofPort> = Arc::new(
        uc_app::usecases::space_access::HmacProofAdapter::new_with_encryption_session(
            deps.security.encryption_session.clone(),
        ),
    );
    let timer_port: Arc<Mutex<dyn TimerPort>> = Arc::new(Mutex::new(uc_infra::time::Timer::new()));
    let persistence_port = Arc::new(Mutex::new(
        uc_app::usecases::space_access::SpaceAccessPersistenceAdapter::new(
            deps.security.encryption_state.clone(),
            deps.device.paired_device_repo.clone(),
            Arc::new(uc_app::usecases::StagedPairedDeviceStore::new()),
        ),
    ));
    let setup_event_port = Arc::new(HostEventSetupPort::new(emitter_cell));  // <-- from separate parameter, NOT ports

    Arc::new(SetupOrchestrator::new(
        initialize_encryption,
        mark_setup_complete,
        deps.setup_status.clone(),
        app_lifecycle,
        ports.pairing_orchestrator,
        setup_event_port,
        ports.space_access_orchestrator,
        ports.discovery_port,
        deps.network_control.clone(),
        crypto_factory,
        deps.network_ports.pairing.clone(),
        transport_port,
        proof_port,
        timer_port,
        persistence_port,
    ))
}
```

4. **Remove build_setup_orchestrator from AppRuntime impl** in runtime.rs. Also remove `SetupRuntimePorts` struct and its impl block. Update `AppRuntime::with_setup()` to create shared state ONCE and pass it to BOTH `build_setup_orchestrator` and `CoreRuntime::new()`:

**CRITICAL SINGLE-SOURCE-OF-TRUTH PATTERN**: `with_setup()` creates `emitter_cell`, `lifecycle_status`, and `clipboard_integration_mode` FIRST. It then passes them to `build_setup_orchestrator()` as separate parameters (NOT inside SetupAssemblyPorts). It then passes the SAME instances to `CoreRuntime::new()`. This guarantees the orchestrator and CoreRuntime share identical state.

```rust
pub fn with_setup(
    deps: AppDeps,
    setup_ports: super::assembly::SetupAssemblyPorts,  // network/external ports only
    watcher_control: Arc<dyn uc_platform::ports::WatcherControlPort>,
    storage_paths: uc_app::app_paths::AppPaths,
    event_emitter: Arc<dyn HostEventEmitterPort>,
) -> Self {
    // 1. Create shared state ONCE — these are the single source of truth
    let lifecycle_status: Arc<dyn uc_app::usecases::LifecycleStatusPort> =
        Arc::new(uc_app::usecases::InMemoryLifecycleStatus::new());
    let clipboard_integration_mode = super::resolve_clipboard_integration_mode();
    let emitter_cell = Arc::new(std::sync::RwLock::new(event_emitter));

    // 2. Create app_handle and session_ready_emitter BEFORE build_setup_orchestrator
    //    (TauriSessionReadyEmitter needs app_handle, which is only available here)
    let app_handle = Arc::new(std::sync::RwLock::new(None));
    let session_ready_emitter: Arc<dyn uc_app::usecases::SessionReadyEmitter> =
        Arc::new(crate::adapters::lifecycle::TauriSessionReadyEmitter::new(
            app_handle.clone(),
        ));

    // 3. Pass shared state + session_ready_emitter to build_setup_orchestrator
    //    as SEPARATE parameters (not inside SetupAssemblyPorts)
    let setup_orchestrator = super::assembly::build_setup_orchestrator(
        &deps,
        setup_ports,
        lifecycle_status.clone(),       // same instance goes to CoreRuntime below
        emitter_cell.clone(),           // same instance goes to CoreRuntime below
        clipboard_integration_mode,     // same value goes to CoreRuntime below
        session_ready_emitter,          // constructed from app_handle above
    );

    // 4. Pass the SAME shared state to CoreRuntime
    let task_registry = Arc::new(uc_app::task_registry::TaskRegistry::new());
    let core = Arc::new(CoreRuntime::new(
        deps,
        emitter_cell,                // same cell as build_setup_orchestrator received
        lifecycle_status,            // same status as build_setup_orchestrator received
        setup_orchestrator,
        clipboard_integration_mode,  // same mode as build_setup_orchestrator received
        task_registry,
        storage_paths,
    ));

    Self { core, app_handle, watcher_control }
}
```

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use std::sync::Mutex as StdMutex;

    struct RecordingEmitter {
        events: Arc<StdMutex<Vec<String>>>,
    }
    impl HostEventEmitterPort for RecordingEmitter {
        fn emit(&self, event: HostEvent) -> Result<(), anyhow::Error> {
            self.events.lock().unwrap().push(format!("{:?}", event));
            Ok(())
        }
    }

    #[tokio::test]
    async fn setup_state_emission_survives_emitter_swap() {
        // 1. Create cell with initial logging emitter
        use uc_app::usecases::app_lifecycle::adapters::LoggingSessionReadyEmitter;
        let initial: Arc<dyn HostEventEmitterPort> =
            // Use a noop emitter — LoggingSessionReadyEmitter doesn't impl HostEventEmitterPort
            // Use the LoggingEventEmitter from uc-tauri adapters
            Arc::new(crate::adapters::host_event_emitter::LoggingEventEmitter);
        let cell = Arc::new(std::sync::RwLock::new(initial));

        // 2. Create HostEventSetupPort with the cell
        let setup_port = HostEventSetupPort::new(cell.clone());

        // 3. Swap emitter to recording emitter
        let events = Arc::new(StdMutex::new(vec![]));
        let recording: Arc<dyn HostEventEmitterPort> =
            Arc::new(RecordingEmitter { events: events.clone() });
        *cell.write().unwrap() = recording;

        // 4. Emit through setup_port — should reach recording emitter
        setup_port.emit_setup_state_changed(
            uc_core::setup::SetupState::FirstRun,
            None,
        ).await;

        // 5. Verify event reached new emitter
        let recorded = events.lock().unwrap();
        assert_eq!(recorded.len(), 1, "Setup event should reach the new emitter after swap");
        assert!(recorded[0].contains("FirstRun"), "Event should contain FirstRun state");
    }
}
```
