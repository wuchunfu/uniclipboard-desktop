    std::fs::read_to_string(resolve_daemon_pid_path())

        .ok()

pub fn remove_pid_file() {
    let _ = std::fs::remove_file(resolve_daemon_pid_path());

use crate::exit_codes;
use crate::pid;

#[cfg(unix)]
pub async fn run(json: bool, _verbose: bool) -> i32 {
    // 1. Check if daemon already running via PID file
    if let Some(existing_pid) = pid::read_pid_file() {
        if pid::is_process_alive(existing_pid) {
            if json {
                println!(r#"{{"status":"already_running","pid":{}}}"#, existing_pid);
            } else {
                eprintln!("Daemon already running (PID {})", existing_pid);
            }
            return exit_codes::EXIT_ERROR;
        }
        // Stale PID file — clean up
        pid::remove_pid_file();
    }

    // 2. Resolve daemon binary path (same directory as CLI binary)
    let daemon_bin = match std::env::current_exe() {
        Ok(exe) => exe.parent()
            .map(|p| p.join("uniclipboard-daemon"))
            .unwrap_or_else(|| "uniclipboard-daemon".into()),
        Err(_) => "uniclipboard-daemon".into(),
    };

    // 3. Spawn daemon as detached process with setsid for proper daemonization
    use std::os::unix::process::CommandExt;
    use std::process::Command;
    let mut cmd = Command::new(&daemon_bin);
    cmd.stdin(std::process::Stdio::null())
        .stdout(std::process::Stdio::null())
        .stderr(std::process::Stdio::null());
    // SAFETY: setsid() is async-signal-safe and creates a new session,
    // detaching the daemon from the CLI's terminal/process group.
    unsafe {
        cmd.pre_exec(|| {
            nix::unistd::setsid()
                .map(|_| ())
                .map_err(|e| std::io::Error::from_raw_os_error(e as i32))
        });
    }
    let child = match cmd.spawn() {
        Ok(child) => child,
        Err(e) => {
            eprintln!("Error: failed to start daemon: {}", e);
            return exit_codes::EXIT_ERROR;
        }
    };

    let child_pid = child.id();

    // 4. Write PID file
    if let Err(e) = pid::write_pid_file(child_pid) {
        eprintln!("Warning: failed to write PID file: {}", e);
    }

    // 5. Brief wait then verify process is alive
    tokio::time::sleep(std::time::Duration::from_millis(500)).await;
    if !pid::is_process_alive(child_pid) {
        pid::remove_pid_file();
        eprintln!("Error: daemon exited immediately after start");
        return exit_codes::EXIT_ERROR;
    }

    if json {
        println!(r#"{{"status":"started","pid":{}}}"#, child_pid);
    } else {
        println!("Daemon started (PID {})", child_pid);
    }
    exit_codes::EXIT_SUCCESS
}

#[cfg(not(unix))]
pub async fn run(_json: bool, _verbose: bool) -> i32 {
    eprintln!("Daemon management not supported on this platform");
    exit_codes::EXIT_ERROR
}

use crate::exit_codes;
use crate::pid;

#[cfg(unix)]
pub async fn run(json: bool, _verbose: bool) -> i32 {
    // 1. Read PID file
    let daemon_pid = match pid::read_pid_file() {
        Some(pid) => pid,
        None => {
            if json {
                println!(r#"{{"status":"not_running"}}"#);
            } else {
                eprintln!("Daemon is not running (no PID file)");
            }
            return exit_codes::EXIT_ERROR;
        }
    };

    if !pid::is_process_alive(daemon_pid) {
        pid::remove_pid_file();
        if json {
            println!(r#"{{"status":"not_running"}}"#);
        } else {
            eprintln!("Daemon is not running (stale PID file cleaned)");
        }
        return exit_codes::EXIT_ERROR;
    }

    // 2. Try graceful shutdown via RPC first
    let rpc_ok = try_rpc_shutdown().await;

    if rpc_ok {
        // Wait up to 5s for process to exit
        for _ in 0..50 {
            if !pid::is_process_alive(daemon_pid) {
                break;
            }
            tokio::time::sleep(std::time::Duration::from_millis(100)).await;
        }
    }

    // 3. If still alive, send SIGTERM
    if pid::is_process_alive(daemon_pid) {
        use nix::sys::signal::{kill, Signal};
        use nix::unistd::Pid;
        let _ = kill(Pid::from_raw(daemon_pid as i32), Signal::SIGTERM);

        // Wait up to 5s for SIGTERM
        for _ in 0..50 {
            if !pid::is_process_alive(daemon_pid) {
                break;
            }
            tokio::time::sleep(std::time::Duration::from_millis(100)).await;
        }
    }

    // 4. Check final state
    if pid::is_process_alive(daemon_pid) {
        eprintln!("Error: daemon (PID {}) did not stop", daemon_pid);
        return exit_codes::EXIT_ERROR;
    }

    pid::remove_pid_file();
    if json {
        println!(r#"{{"status":"stopped","pid":{}}}"#, daemon_pid);
    } else {
        println!("Daemon stopped (PID {})", daemon_pid);
    }
    exit_codes::EXIT_SUCCESS
}

/// Attempt graceful shutdown via RPC. Returns true if RPC call succeeded.
#[cfg(unix)]
async fn try_rpc_shutdown() -> bool {
    use std::time::Duration;
    use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
    use tokio::net::UnixStream;
    use uc_daemon::rpc::types::RpcRequest;
    use uc_daemon::socket::resolve_daemon_socket_path;

    let socket_path = resolve_daemon_socket_path();
    let stream = match tokio::time::timeout(
        Duration::from_secs(2),
        UnixStream::connect(&socket_path),
    ).await {
        Ok(Ok(s)) => s,
        _ => return false,
    };

    let (reader, mut writer) = stream.into_split();
    let request = RpcRequest {
        jsonrpc: "2.0".to_string(),
        method: "shutdown".to_string(),
        id: Some(1),
    };
    let request_json = match serde_json::to_string(&request) {
        Ok(j) => j,
        Err(_) => return false,
    };
    if writer.write_all(format!("{}\n", request_json).as_bytes()).await.is_err() {
        return false;
    }
    let _ = writer.flush().await;

    // Read response (best-effort)
    let mut buf = BufReader::new(reader);
    let mut line = String::new();
    let _ = tokio::time::timeout(Duration::from_secs(2), buf.read_line(&mut line)).await;
    true
}

#[cfg(not(unix))]
pub async fn run(_json: bool, _verbose: bool) -> i32 {
    eprintln!("Daemon management not supported on this platform");
    exit_codes::EXIT_ERROR
}

    }

    // Start fresh

}
