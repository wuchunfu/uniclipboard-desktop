Wire the GUI side to receive clipboard events from the daemon via DaemonWsBridge and switch GUI to Passive clipboard mode.

    ```rust
    async fn run_clipboard_realtime_consumer(
        bridge: Arc<DaemonWsBridge>,
        emitter: Arc<dyn HostEventEmitterPort>,
    ) -> Result<()> {
        let mut rx = bridge.subscribe("clipboard_realtime_consumer", &[RealtimeTopic::Clipboard]).await?;
        run_clipboard_realtime_consumer_with_rx(&mut rx, emitter).await
    }

    async fn run_clipboard_realtime_consumer_with_rx(
        rx: &mut mpsc::Receiver<RealtimeEvent>,
        emitter: Arc<dyn HostEventEmitterPort>,
    ) -> Result<()> {
        while let Some(event) = rx.recv().await {
            match event {
                RealtimeEvent::ClipboardNewContent(e) => {
                    let origin_kind = match e.origin.as_str() {
                        "remote" => ClipboardOriginKind::Remote,
                        _ => ClipboardOriginKind::Local,
                    };
                    let _ = emitter.emit(HostEvent::Clipboard(ClipboardHostEvent::NewContent {
                        entry_id: e.entry_id,
                        preview: e.preview,
                        origin: origin_kind,
                    }));
                }
                _ => {} // ignore non-clipboard events on this subscription
            }
        }
        Ok(())
    }
    ```
