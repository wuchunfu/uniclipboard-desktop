```
[4 bytes]  magic: 0x55 0x43 0x32 0x00  ("UC2\0")
[16 bytes] transfer_id (UUID v4 raw bytes)
[4 bytes]  total_chunks (u32 LE)
[4 bytes]  chunk_size_hint (u32 LE) — nominal chunk size; last chunk may be smaller
[4 bytes]  total_plaintext_len (u32 LE)
-- then for each chunk i in 0..total_chunks:
    [4 bytes]  chunk_ciphertext_len (u32 LE) — length of ciphertext for this chunk
    [N bytes]  ciphertext (plaintext_chunk + 16-byte Poly1305 tag)
```

```rust
//! Chunk-level AEAD encoder and decoder for V2 clipboard wire transfer.
//!
//! # Wire Format
//! See CONTEXT.md § "Blob storage format" for the binary layout.
//! Network transfer format uses sequential chunks without an index table.

use chacha20poly1305::{
    aead::{Aead, Payload},
    KeyInit, XChaCha20Poly1305, XNonce,
};
use uc_core::security::{aad, model::MasterKey};

/// Nominal chunk size: 256 KB.
/// Memory usage per chunk: ~512 KB (read buffer + ciphertext buffer).
pub const CHUNK_SIZE: usize = 256 * 1024;

/// Magic bytes identifying a V2 chunked clipboard payload.
/// "UC2\0" (0x55, 0x43, 0x32, 0x00)
pub const V2_MAGIC: [u8; 4] = [0x55, 0x43, 0x32, 0x00];

#[derive(Debug, thiserror::Error)]
pub enum ChunkedTransferError {
    #[error("invalid V2 magic bytes")]
    InvalidMagic,
    #[error("payload too short to contain valid header")]
    TruncatedHeader,
    #[error("chunk count mismatch: expected {expected}, got {actual}")]
    ChunkCountMismatch { expected: u32, actual: u32 },
    #[error("chunk ciphertext too short for AEAD tag")]
    TruncatedChunk,
    #[error("AEAD decryption failed for chunk {chunk_index}")]
    DecryptFailed { chunk_index: u32 },
    #[error("encryption failed: {0}")]
    EncryptFailed(String),
    #[error("total plaintext length mismatch")]
    LengthMismatch,
}

pub struct ChunkedEncoder;
pub struct ChunkedDecoder;

impl ChunkedEncoder {
    /// Encode `plaintext` into V2 binary wire format.
    ///
    /// # Arguments
    /// * `master_key` — 32-byte XChaCha20-Poly1305 key
    /// * `transfer_id` — 16-byte UUID v4 (raw bytes), uniquely identifies this transfer
    /// * `plaintext` — raw bytes to encode (e.g., serde_json of ClipboardMultiRepPayloadV2)
    pub fn encode(
        master_key: &MasterKey,
        transfer_id: &[u8; 16],
        plaintext: &[u8],
    ) -> Result<Vec<u8>, ChunkedTransferError> {
        let cipher = XChaCha20Poly1305::new_from_slice(master_key.as_bytes())
            .map_err(|e| ChunkedTransferError::EncryptFailed(e.to_string()))?;

        let total_plaintext_len = plaintext.len() as u32;
        let chunk_size = CHUNK_SIZE;
        let chunks: Vec<&[u8]> = if plaintext.is_empty() {
            vec![]
        } else {
            plaintext.chunks(chunk_size).collect()
        };
        let total_chunks = chunks.len() as u32;

        // Pre-allocate: header (28 bytes) + each chunk (4 + len + 16 overhead)
        let estimated_size = 28 + plaintext.len() + chunks.len() * (4 + 16);
        let mut out = Vec::with_capacity(estimated_size);

        // Header
        out.extend_from_slice(&V2_MAGIC);
        out.extend_from_slice(transfer_id);
        out.extend_from_slice(&total_chunks.to_le_bytes());
        out.extend_from_slice(&(chunk_size as u32).to_le_bytes());
        out.extend_from_slice(&total_plaintext_len.to_le_bytes());

        // Chunks
        for (i, chunk_plaintext) in chunks.iter().enumerate() {
            let chunk_index = i as u32;
            let nonce_bytes = derive_chunk_nonce(transfer_id, chunk_index);
            let aad_bytes = aad::for_chunk_transfer(transfer_id, chunk_index);
            let ciphertext = cipher
                .encrypt(
                    XNonce::from_slice(&nonce_bytes),
                    Payload { msg: chunk_plaintext, aad: &aad_bytes },
                )
                .map_err(|e| ChunkedTransferError::EncryptFailed(e.to_string()))?;

            out.extend_from_slice(&(ciphertext.len() as u32).to_le_bytes());
            out.extend_from_slice(&ciphertext);
        }

        Ok(out)
    }
}

impl ChunkedDecoder {
    /// Decode V2 binary wire format back to plaintext.
    ///
    /// Validates magic, decrypts each chunk sequentially, verifies AEAD tags.
    /// On any chunk failure, returns Err immediately (no partial data).
    pub fn decode(
        master_key: &MasterKey,
        encoded: &[u8],
    ) -> Result<Vec<u8>, ChunkedTransferError> {
        // Parse header (28 bytes minimum)
        if encoded.len() < 28 {
            return Err(ChunkedTransferError::TruncatedHeader);
        }
        if &encoded[0..4] != &V2_MAGIC {
            return Err(ChunkedTransferError::InvalidMagic);
        }
        let transfer_id: [u8; 16] = encoded[4..20].try_into().expect("16 bytes");
        let total_chunks = u32::from_le_bytes(encoded[20..24].try_into().expect("4 bytes"));
        // chunk_size_hint at [24..28] — read for validation but not strictly needed for decode
        let _chunk_size_hint = u32::from_le_bytes(encoded[24..28].try_into().expect("4 bytes"));
        // total_plaintext_len at [28..32]
        if encoded.len() < 32 {
            return Err(ChunkedTransferError::TruncatedHeader);
        }
        let total_plaintext_len = u32::from_le_bytes(encoded[28..32].try_into().expect("4 bytes")) as usize;

        let cipher = XChaCha20Poly1305::new_from_slice(master_key.as_bytes())
            .map_err(|e| ChunkedTransferError::EncryptFailed(e.to_string()))?;

        let mut plaintext = Vec::with_capacity(total_plaintext_len);
        let mut pos = 32usize; // after fixed header
        let mut decoded_chunks = 0u32;

        while decoded_chunks < total_chunks {
            if pos + 4 > encoded.len() {
                return Err(ChunkedTransferError::TruncatedHeader);
            }
            let ciphertext_len = u32::from_le_bytes(
                encoded[pos..pos + 4].try_into().expect("4 bytes")
            ) as usize;
            pos += 4;

            if pos + ciphertext_len > encoded.len() {
                return Err(ChunkedTransferError::TruncatedChunk);
            }
            let ciphertext = &encoded[pos..pos + ciphertext_len];
            pos += ciphertext_len;

            let chunk_index = decoded_chunks;
            let nonce_bytes = derive_chunk_nonce(&transfer_id, chunk_index);
            let aad_bytes = aad::for_chunk_transfer(&transfer_id, chunk_index);

            let chunk_plaintext = cipher
                .decrypt(
                    XNonce::from_slice(&nonce_bytes),
                    Payload { msg: ciphertext, aad: &aad_bytes },
                )
                .map_err(|_| ChunkedTransferError::DecryptFailed { chunk_index })?;

            plaintext.extend_from_slice(&chunk_plaintext);
            decoded_chunks += 1;
        }

        Ok(plaintext)
    }
}

/// Derive a 24-byte XChaCha20 nonce for a chunk.
///
/// `nonce = blake3("uc:chunk-nonce:v1|" || transfer_id || chunk_index_le)[0..24]`
fn derive_chunk_nonce(transfer_id: &[u8; 16], chunk_index: u32) -> [u8; 24] {
    let mut hasher = blake3::Hasher::new();
    hasher.update(b"uc:chunk-nonce:v1|");
    hasher.update(transfer_id);
    hasher.update(&chunk_index.to_le_bytes());
    let hash = hasher.finalize();
    let mut nonce = [0u8; 24];
    nonce.copy_from_slice(&hash.as_bytes()[..24]);
    nonce
}
```
