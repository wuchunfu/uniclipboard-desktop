    ```rust
            async move {

                            payload: serde_json::to_value(payload).unwrap_or_default(),
    ```

    ```rust
    #[async_trait]
    impl DaemonService for ClipboardWatcherWorker {
        fn name(&self) -> &str { "clipboard-watcher" }

        async fn start(&self, cancel: CancellationToken) -> anyhow::Result<()> {
            // Create mpsc channel for ClipboardWatcher -> async handler bridge
            let (platform_tx, mut platform_rx) = tokio::sync::mpsc::channel(64);

            // Create ClipboardWatcherContext and register handler
            let mut watcher_ctx = ClipboardWatcherContext::new()
                .map_err(|e| anyhow::anyhow!("Failed to create watcher context: {}", e))?;
            let watcher = ClipboardWatcher::new(self.local_clipboard.clone(), platform_tx);
            let shutdown = watcher_ctx.add_handler(watcher).get_shutdown_channel();

            // Spawn blocking watcher thread (per D-07)
            let _watcher_join = tokio::task::spawn_blocking(move || {
                info!("daemon clipboard watcher started");
                watcher_ctx.start_watch();
                info!("daemon clipboard watcher stopped");
            });

            // Event processing loop with cancellation support (per D-08)
            loop {
                tokio::select! {
                    _ = cancel.cancelled() => {
                        info!("daemon clipboard watcher cancelling");
                        shutdown.stop();
                        break;
                    }
                    event = platform_rx.recv() => {
                        match event {
                            Some(PlatformEvent::ClipboardChanged { snapshot }) => {
                                if snapshot.is_empty() {
                                    debug!("empty clipboard snapshot, skipping");
                                    continue;
                                }
                                if let Err(e) = self.change_handler.on_clipboard_changed(snapshot).await {
                                    warn!(error = %e, "clipboard change handler failed");
                                }
                            }
                            Some(_) => {} // Ignore other platform events
                            None => {
                                warn!("platform event channel closed unexpectedly");
                                break;
                            }
                        }
                    }
                }
            }
            Ok(())
        }

        async fn stop(&self) -> anyhow::Result<()> {
            info!("clipboard watcher service stopped");
            Ok(())
        }

        fn health_check(&self) -> ServiceHealth {
            ServiceHealth::Healthy
        }
    }
    ```
