```rust
use serde::{Serialize, Deserialize};
use crate::worker::WorkerHealth;

#[derive(Debug, Serialize, Deserialize)]
pub struct RpcRequest {
    pub jsonrpc: String,
    pub method: String,
    pub id: Option<u64>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct RpcResponse {
    pub jsonrpc: String,
    pub result: Option<serde_json::Value>,
    pub error: Option<RpcError>,
    pub id: Option<u64>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct RpcError {
    pub code: i32,
    pub message: String,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct StatusResponse {
    pub uptime_seconds: u64,
    pub version: String,
    pub workers: Vec<WorkerStatus>,
    pub connected_peers: Option<u32>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct WorkerStatus {
    pub name: String,
    pub health: WorkerHealth,
}

impl RpcResponse {
    pub fn success(id: Option<u64>, result: serde_json::Value) -> Self {
        Self { jsonrpc: "2.0".to_string(), result: Some(result), error: None, id }
    }
    pub fn error(id: Option<u64>, code: i32, message: String) -> Self {
        Self { jsonrpc: "2.0".to_string(), result: None, error: Some(RpcError { code, message }), id }
    }
}
```

**9. Create `src-tauri/crates/uc-daemon/src/state.rs`:**

```rust
use std::time::Instant;
use crate::worker::{DaemonWorker, WorkerHealth};
use crate::rpc::types::WorkerStatus;

pub struct RuntimeState {
    start_time: Instant,
    workers: Vec<Box<dyn DaemonWorker>>,
}

impl RuntimeState {
    pub fn new(workers: Vec<Box<dyn DaemonWorker>>) -> Self {
        Self { start_time: Instant::now(), workers }
    }
    pub fn uptime_seconds(&self) -> u64 {
        self.start_time.elapsed().as_secs()
    }
    pub fn worker_statuses(&self) -> Vec<WorkerStatus> {
        self.workers.iter().map(|w| WorkerStatus {
            name: w.name().to_string(),
            health: w.health_check(),
        }).collect()
    }
    pub fn workers(&self) -> &[Box<dyn DaemonWorker>] {
        &self.workers
    }
    pub fn workers_mut(&mut self) -> &mut Vec<Box<dyn DaemonWorker>> {
        &mut self.workers
    }
}
```
