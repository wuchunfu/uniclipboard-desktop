Create `.github/workflows/build-cli.yml` with the following structure:

```yaml
name: 'Build CLI'
run-name: "build-cli-${{ inputs.platform || 'all' }}-${{ github.run_number }}"
on:
  workflow_dispatch:
    inputs:
      platform:
        description: 'Select build platform'
        required: true
        type: choice
        options:
          - 'macos-aarch64'
          - 'macos-x86_64'
          - 'ubuntu-22.04'
          - 'windows-latest'
          - 'all'
        default: 'all'
  workflow_call:
    inputs:
      platform:
        description: 'Select build platform'
        required: false
        type: string
        default: 'all'
      branch:
        description: 'Branch to build'
        required: false
        type: string
        default: ''
```

**Jobs:**

1.  `setup-matrix` — identical to build.yml's setup-matrix job (same JSON matrix with platform/target/args).

2.  `build-cli`:
    - `name: Build CLI ${{ matrix.target }}`
    - `needs: setup-matrix`
    - `runs-on: ${{ matrix.platform }}`
    - `strategy.fail-fast: false`
    - `strategy.matrix.include: ${{ fromJSON(needs.setup-matrix.outputs.matrix) }}`

    Steps:
    a. `actions/checkout@v4` with `ref: ${{ inputs.branch || github.ref }}`
    b. `actions/setup-node@v4` with `node-version: lts/*` (needed for `package.json` version read)
    c. `dtolnay/rust-toolchain@stable` with targets for macOS cross-compilation:
    `targets: ${{ matrix.platform == 'macos-latest' && 'aarch64-apple-darwin,x86_64-apple-darwin' || '' }}`
    d. `Swatinem/rust-cache@v2` with: - `workspaces: "src-tauri -> target"` - `shared-key: cli-${{ matrix.target }}`
    e. Linux deps install (if ubuntu-22.04):
    `       sudo apt-get update
  sudo apt-get install -y libwebkit2gtk-4.1-dev libappindicator3-dev librsvg2-dev patchelf
  `
    f. Build CLI binary:
    `       working-directory: src-tauri
  run: cargo build --release -p uc-cli ${{ matrix.args }}
  `
    g. Package CLI binary (shell: bash):

    ````bash
    VERSION=$(node -p "require('./package.json').version")
      TARGET="${{ matrix.target }}"
    BIN_NAME="uniclipboard-cli"

          if [ "${{ matrix.platform }}" = "windows-latest" ]; then
            BIN_SRC="src-tauri/target/release/${BIN_NAME}.exe"
            # Windows non-cross build puts binary in target/release/, not target/{target}/release/
            if [ ! -f "$BIN_SRC" ]; then
              BIN_SRC="src-tauri/target/${TARGET}/release/${BIN_NAME}.exe"
            fi
            ARCHIVE="uniclipboard-cli-${VERSION}-${TARGET}.zip"
            7z a "$ARCHIVE" "$BIN_SRC"
          else
            BIN_DIR="src-tauri/target/${TARGET}/release"
            # Fallback for non-cross builds
            if [ ! -f "${BIN_DIR}/${BIN_NAME}" ]; then
              BIN_DIR="src-tauri/target/release"
            fi
            ARCHIVE="uniclipboard-cli-${VERSION}-${TARGET}.tar.gz"
            tar -czf "$ARCHIVE" -C "$BIN_DIR" "$BIN_NAME"
          fi
          echo "archive=$ARCHIVE" >> "$GITHUB_OUTPUT"
          ```

    h. Upload artifact:
    `yaml
      uses: actions/upload-artifact@v4
      with:
        name: cli-${{ matrix.target }}
        path: uniclipboard-cli-*
      `
    </action>
    ````
