    For clipboard events, we need a simpler consumer that:

    2. Receives `RealtimeEvent::ClipboardNewContent`

    Create this consumer function in `src-tauri/crates/uc-daemon-client/src/realtime.rs` (or nearby):

    ```rust
    pub async fn run_clipboard_realtime_consumer(
        bridge: &DaemonWsBridge,
        emitter: Arc<dyn HostEventEmitterPort>,
        cancel: CancellationToken,
    ) {
        let Ok(mut rx) = bridge.subscribe("clipboard-consumer", &[RealtimeTopic::Clipboard]).await else {
            warn!("failed to subscribe to clipboard realtime topic");
            return;
        };

        loop {
            tokio::select! {
                _ = cancel.cancelled() => break,
                Some(event) = rx.recv() => {
                    if let RealtimeEvent::ClipboardNewContent(ev) = event {
                        let origin = match ev.origin.as_str() {
                            "remote" => ClipboardOriginKind::Remote,
                            _ => ClipboardOriginKind::Local,
                        };
                        let _ = emitter.emit(HostEvent::Clipboard(
                            ClipboardHostEvent::NewContent {
                                entry_id: ev.entry_id,
                                preview: ev.preview,
                                origin,
                            },
                        ));
                    }
                }
            }
        }
    }
    ```

    Then wire this consumer into `start_realtime_runtime` alongside the existing pairing/peers/setup consumers:
    ```rust
    let clipboard_cancel = cancel.child_token();
    let clipboard_emitter = emitter.clone();
    task_registry.spawn("clipboard-realtime", async move {
        run_clipboard_realtime_consumer(&bridge, clipboard_emitter, clipboard_cancel).await;
    });
    ```
