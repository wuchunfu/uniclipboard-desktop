   ```rust
   use std::sync::Arc;
   use anyhow::Result;
   use tracing::{info, info_span, warn, Instrument};

   /// Use case: Clean up expired file cache entries
   ///
   /// Runs on app startup to:
   /// 1. Query file entries older than retention period with status Completed
   /// 2. Query file entries with status Failed (cleanup immediately)
   /// 3. Delete associated temp files from disk
   /// 4. Update database status to Expired
   /// 5. Log summary: files removed, space reclaimed
   pub struct CleanupExpiredFilesUseCase {
       settings_port: Arc<dyn SettingsPort>,
       file_entry_repo: Arc<dyn FileEntryRepository>,
   }

   impl CleanupExpiredFilesUseCase {
       pub fn new(
           settings_port: Arc<dyn SettingsPort>,
           file_entry_repo: Arc<dyn FileEntryRepository>,
       ) -> Self {
           Self { settings_port, file_entry_repo }
       }

       pub async fn execute(&self) -> Result<CleanupResult> {
           let span = info_span!("usecase.cleanup_expired_files.execute");
           async {
               let settings = self.settings_port.get_settings().await?;

               if !settings.sync.file_auto_cleanup {
                   info!("Auto-cleanup disabled, skipping");
                   return Ok(CleanupResult::default());
               }

               let retention_hours = settings.sync.file_retention_hours;
               let cutoff = chrono::Utc::now() - chrono::Duration::hours(retention_hours as i64);

               // Query expired entries: completed + older than retention period
               let expired_entries = self.file_entry_repo
                   .find_expired_entries(cutoff)
                   .instrument(info_span!("query_expired_entries"))
                   .await?;

               // Query failed entries: cleanup immediately regardless of age
               let failed_entries = self.file_entry_repo
                   .find_failed_entries()
                   .instrument(info_span!("query_failed_entries"))
                   .await?;

               let all_entries: Vec<_> = expired_entries.into_iter()
                   .chain(failed_entries.into_iter())
                   .collect();

               if all_entries.is_empty() {
                   info!("No expired files to clean up");
                   return Ok(CleanupResult::default());
               }

               let mut files_removed = 0u32;
               let mut bytes_reclaimed = 0u64;
               let mut errors = 0u32;

               for entry in &all_entries {
                   if let Some(cached_path) = &entry.cached_path {
                       match tokio::fs::remove_file(cached_path).await {
                           Ok(()) => {
                               files_removed += 1;
                               bytes_reclaimed += entry.file_size;
                           }
                           Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
                               // File already gone, still mark as expired in DB
                               files_removed += 1;
                           }
                           Err(e) => {
                               warn!(path = %cached_path.display(), error = %e, "Failed to delete cached file");
                               errors += 1;
                           }
                       }
                   }

                   // Update status to Expired in database
                   if let Err(e) = self.file_entry_repo
                       .update_status(entry.id, FileTransferStatus::Expired)
                       .await
                   {
                       warn!(entry_id = %entry.id, error = %e, "Failed to update file entry status");
                       errors += 1;
                   }
               }

               let result = CleanupResult { files_removed, bytes_reclaimed, errors };
               info!(
                   files_removed = result.files_removed,
                   bytes_reclaimed_mb = result.bytes_reclaimed / (1024 * 1024),
                   errors = result.errors,
                   "File cache cleanup complete"
               );

               Ok(result)
           }.instrument(span).await
       }
   }

   #[derive(Debug, Default)]
   pub struct CleanupResult {
       pub files_removed: u32,
       pub bytes_reclaimed: u64,
       pub errors: u32,
   }
   ```

3. Add a `check_device_quota` utility function in the same file or a sibling `quota.rs`:

   ```rust
   /// Check if accepting a file would exceed the per-device cache quota.
   ///
   /// Returns Ok(()) if within quota, Err with quota details if exceeded.
   pub async fn check_device_quota(
       file_entry_repo: &dyn FileEntryRepository,
       settings_port: &dyn SettingsPort,
       source_device_id: &str,
       incoming_file_size: u64,
   ) -> Result<(), QuotaExceededError> {
       let settings = settings_port.get_settings().await
           .map_err(|e| QuotaExceededError::Internal(e.to_string()))?;

       let quota_bytes = settings.sync.file_cache_quota_per_device_mb as u64 * 1024 * 1024;

       let current_usage = file_entry_repo
           .calculate_device_cache_usage(source_device_id)
           .await
           .map_err(|e| QuotaExceededError::Internal(e.to_string()))?;

       if current_usage + incoming_file_size > quota_bytes {
           return Err(QuotaExceededError::Exceeded {
               device_id: source_device_id.to_string(),
               current_usage,
               quota: quota_bytes,
               requested: incoming_file_size,
           });
       }

       Ok(())
   }

   #[derive(Debug, thiserror::Error)]
   pub enum QuotaExceededError {
       #[error("Cache quota exceeded for device {device_id}: {current_usage}/{quota} bytes used, requested {requested} bytes")]
       Exceeded {
           device_id: String,
           current_usage: u64,
           quota: u64,
           requested: u64,
       },
       #[error("Internal error checking quota: {0}")]
       Internal(String),
   }
   ```
