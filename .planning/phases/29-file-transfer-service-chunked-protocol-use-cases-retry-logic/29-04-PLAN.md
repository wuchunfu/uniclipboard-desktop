    ```rust
    #[async_trait]
    impl FileTransportPort for Libp2pNetworkAdapter {
        async fn send_file_announce(&self, _peer_id: &str, _announce: FileTransferMessage) -> Result<()> {
            // Individual message methods are not used — full transfer goes through send_file()
            Ok(())
        }
        async fn send_file_data(&self, _peer_id: &str, _data: FileTransferMessage) -> Result<()> { Ok(()) }
        async fn send_file_complete(&self, _peer_id: &str, _complete: FileTransferMessage) -> Result<()> { Ok(()) }
        async fn cancel_transfer(&self, _peer_id: &str, _cancel: FileTransferMessage) -> Result<()> { Ok(()) }

        async fn send_file(&self, peer_id: &str, file_path: PathBuf, transfer_id: String,
                           batch_id: Option<String>, batch_total: Option<u32>) -> Result<()> {
            let service = self.file_transfer_service.lock()
                .map_err(|_| anyhow!("file transfer service mutex poisoned"))?;
            let service = service.as_ref()
                .ok_or_else(|| anyhow!("file transfer service not initialized — network not started"))?;
            service.send_file(peer_id, file_path, transfer_id, batch_id, batch_total).await
        }
    }
    ```

    ```rust
                .map_err(|_| anyhow!("..."))?;
            guard.as_ref().ok_or_else(|| anyhow!("..."))?.clone()
        };
        service.send_file(peer_id, file_path, transfer_id, batch_id, batch_total).await
    }
    ```
