use crate::exit_codes;
use std::io::{BufRead, BufReader, Seek, SeekFrom};

/// Run the daemon logs command.
///
/// Reads the most recent log file from the platform-specific logs directory.
/// Supports --lines N (default 50) and --follow for live tailing.
pub async fn run(lines: usize, follow: bool, json_output: bool, _verbose: bool) -> i32 {
    // 1. Resolve logs directory via platform adapter
    let logs_dir = match resolve_logs_dir() {
        Some(d) => d,
        None => {
            eprintln!("Error: could not resolve application data directory");
            return exit_codes::EXIT_ERROR;
        }
    };

    if !logs_dir.exists() {
        eprintln!("Error: logs directory does not exist: {}", logs_dir.display());
        return exit_codes::EXIT_ERROR;
    }

    // 2. Find most recent log file (uniclipboard.json.YYYY-MM-DD)
    let log_file = match find_latest_log_file(&logs_dir) {
        Some(f) => f,
        None => {
            eprintln!("Error: no log files found in {}", logs_dir.display());
            return exit_codes::EXIT_ERROR;
        }
    };

    if json_output {
        println!(r#"{{"log_file":"{}"}}"#, log_file.display());
    } else {
        eprintln!("Reading: {}", log_file.display());
    }

    // 3. Read last N lines
    let file = match std::fs::File::open(&log_file) {
        Ok(f) => f,
        Err(e) => {
            eprintln!("Error: failed to open log file: {}", e);
            return exit_codes::EXIT_ERROR;
        }
    };

    let reader = BufReader::new(&file);
    let all_lines: Vec<String> = reader.lines().filter_map(|l| l.ok()).collect();
    let start = all_lines.len().saturating_sub(lines);
    for line in &all_lines[start..] {
        println!("{}", line);
    }

    // 4. Follow mode: poll for new lines
    if follow {
        use std::time::Duration;
        let mut file = match std::fs::File::open(&log_file) {
            Ok(f) => f,
            Err(e) => {
                eprintln!("Error: failed to reopen log file: {}", e);
                return exit_codes::EXIT_ERROR;
            }
        };
        // Seek to end
        let _ = file.seek(SeekFrom::End(0));
        let mut reader = BufReader::new(file);
        loop {
            let mut line = String::new();
            match reader.read_line(&mut line) {
                Ok(0) => {
                    // No new data, sleep briefly
                    tokio::time::sleep(Duration::from_millis(200)).await;
                }
                Ok(_) => {
                    print!("{}", line);
                }
                Err(e) => {
                    eprintln!("Error reading log: {}", e);
                    return exit_codes::EXIT_ERROR;
                }
            }
        }
    }

    exit_codes::EXIT_SUCCESS
}

fn resolve_logs_dir() -> Option<std::path::PathBuf> {
    use uc_platform::app_dirs::DirsAppDirsAdapter;
    use uc_platform::ports::AppDirsPort;
    let adapter = DirsAppDirsAdapter::new();
    let app_dirs = adapter.get_app_dirs().ok()?;
    Some(app_dirs.app_data_root.join("logs"))
}

fn find_latest_log_file(logs_dir: &std::path::Path) -> Option<std::path::PathBuf> {
    let mut entries: Vec<_> = std::fs::read_dir(logs_dir)
        .ok()?
        .filter_map(|e| e.ok())
        .filter(|e| {
            e.file_name()
                .to_str()
                .map(|n| n.starts_with("uniclipboard.json"))
                .unwrap_or(false)
        })
        .collect();
    // Sort by name descending (YYYY-MM-DD suffix sorts chronologically)
    entries.sort_by(|a, b| b.file_name().cmp(&a.file_name()));
    entries.first().map(|e| e.path())
}
