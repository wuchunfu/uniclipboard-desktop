**6. Create `src-tauri/crates/uc-cli/src/main.rs`:**

```rust
mod exit_codes;
mod output;
mod commands;

use clap::{Parser, Subcommand};

#[derive(Parser)]
#[command(name = "uniclipboard-cli", version, about = "UniClipboard command-line interface")]
struct Cli {
    /// Output in JSON format
    #[arg(long, global = true)]
    json: bool,

    #[command(subcommand)]
    command: Commands,
}

#[derive(Subcommand)]
enum Commands {
    /// Show daemon status
    Status,
    /// List paired devices (direct mode, no daemon required)
    Devices,
    /// Show space and encryption status (direct mode, no daemon required)
    SpaceStatus,
}

fn main() {
    let cli = Cli::parse();

    let rt = tokio::runtime::Builder::new_multi_thread()
        .enable_all()
        .build()
        .expect("Failed to create Tokio runtime");

    let exit_code = rt.block_on(async {
        match cli.command {
            Commands::Status => {
                // Resolve socket path — need config for data dir
                // Use uc-bootstrap config resolution to get app data dir
                // Then: {data_dir}/uniclipboard-daemon.sock
                let socket_path = match resolve_socket_path() {
                    Ok(p) => p,
                    Err(e) => {
                        eprintln!("Error: failed to resolve daemon socket path: {}", e);
                        return exit_codes::EXIT_ERROR;
                    }
                };
                commands::status::run(&socket_path, cli.json).await
            }
            Commands::Devices => {
                commands::devices::run(cli.json).await
            }
            Commands::SpaceStatus => {
                commands::space_status::run(cli.json).await
            }
        }
    });

    std::process::exit(exit_code);
}

fn resolve_socket_path() -> anyhow::Result<std::path::PathBuf> {
    // Minimal bootstrap: just resolve config to get data dir
    // Use resolve_app_config() + get_storage_paths() from uc-bootstrap
    let config = uc_bootstrap::resolve_app_config()
        .map_err(|e| anyhow::anyhow!("{}", e))?;
    let paths = uc_bootstrap::get_storage_paths(&config)?;
    Ok(paths.data_dir.join("uniclipboard-daemon.sock"))
}
```
