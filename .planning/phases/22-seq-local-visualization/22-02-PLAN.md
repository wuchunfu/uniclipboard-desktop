1. **Update tracing.rs** to compose the Seq layer:
   - Add import: `use uc_observability::SeqGuard;` (build_seq_layer is accessed via `uc_observability::build_seq_layer`)
   - Add static: `static SEQ_GUARD: OnceLock<SeqGuard> = OnceLock::new();`
   - After existing `let (json_layer, guard) = ...` and before the registry composition, add:
     ```rust
     // Step 4b: Build Seq layer (if UC_SEQ_URL is set)
     let seq_enabled;
     let seq_layer = match uc_observability::build_seq_layer(&profile) {
         Some((layer, guard)) => {
             seq_enabled = true;
             if SEQ_GUARD.set(guard).is_err() {
                 eprintln!("Seq guard already initialized");
             }
             Some(layer)
         }
         None => {
             seq_enabled = false;
             None
         }
     };
     ```
   - Update the registry composition to include `.with(seq_layer)` after `.with(json_layer)`
   - Update the final info! log message to indicate Seq status:
     ```rust
     tracing::info!(
         profile = %profile,
         logs_dir = %paths.logs_dir.display(),
         seq_enabled = seq_enabled,
         "Tracing initialized with dual output (console + JSON{})",
         if seq_enabled { " + Seq" } else { "" }
     );
     ```
   - Note: The `seq_layer` variable will be `Option<impl Layer>` which works with `.with()` since tracing-subscriber implements `Layer` for `Option<L>`.

2. **Create docker-compose.seq.yml** in project root:

   ```yaml
   # Local Seq instance for UniClipboard log visualization
   # Usage: docker compose -f docker-compose.seq.yml up -d
   # UI: http://localhost:5341
   # Ingestion: http://localhost:5341/ingest/clef
   services:
     seq:
       image: datalust/seq:2024.3
       restart: unless-stopped
       environment:
         ACCEPT_EULA: 'Y'
       ports:
         - '5341:80'
       volumes:
         - seq-data:/data
   volumes:
     seq-data:
   ```
