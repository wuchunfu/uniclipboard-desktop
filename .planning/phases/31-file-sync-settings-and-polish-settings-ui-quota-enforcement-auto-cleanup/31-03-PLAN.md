   ```rust
   if let Err(quota_err) = check_device_quota(

       announce.file_size,
   ).await {
       warn!(%quota_err, "Rejecting file transfer: quota exceeded");
       // Notify sender about rejection

           &source_device,
           &format!("Cache quota exceeded on receiver device: {}", quota_err),
       ).await?;
       // Emit local event for receiver notification
       let _ = self.event_tx.send(NetworkEvent::FileTransferFailed {
           reason: format!(
               "Rejected {} from {}: per-device cache quota exceeded",
               announce.filename, source_device_id,

       return Ok(());
   }
   ```
