        for rep in &normalized_reps {
            if rep.payload_state() == PayloadAvailability::Staged {
                if let Some(observed) =
                    snapshot.representations.iter().find(|o| o.id == rep.id)
                {
                    if let Err(err) = self
                        .spool_queue
                        .enqueue(SpoolRequest {
